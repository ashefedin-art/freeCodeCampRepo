def arithmetic_arranger(problems, show_answers=False):

    # 1.  Check if there are too many problems.
    if len(problems) > 5:
        return 'Error: Too many problems.'
    
    # Lists to hold the formatted parts on each line.
    first_operands = []
    second_operands = []
    operators = []


    for problem in problems:
        parts = problem.split()
        num1_str, operator, num2_str = parts

        #2 check for valid operators.
        if operator not in ['+', '-']:
            return "Error: Operator must be '+' or '-'."
        
        #3 Check if operands are digits.
        if not num1_str.isdigit() or not num2_str.isdigit():
            return 'Error: Numbers must only contain digits.'
        
        #4 check if operants are too long
        if len(num1_str) > 4 or len(num2_str) > 4:
            return 'Error: Numbers cannot be more than four digits.'

        first_operands.append(num1_str)
        second_operands.append(num2_str)
        operators.append(operator)

    # ---- Formatting ----

    line1 = ''
    line2 = ''
    line3 = ''
    line4 = ''

    for i in range(len(problems)):
        num1 = first_operands[i]
        num2 = second_operands[i]
        op = operators[i]

        #determine the width needed for the current problem.

        max_len = max(len(num1), len(num2))
        width = max_len + 2

        # Build each line for the current problem.
        # rjust() right-aligns the strings.

        l1 = num1.rjust(width)
        l2 = op + num2.rjust(width - 1)
        l3 = '-' * width

        # calc the answer if needed
        if show_answers:
            if op == '+':
                answer = str(int(num1) + int(num2))
            else:
                answer = str(int(num1) - int(num2))
            l4 = answer.rjust(width)
        
        # add the formatted parts to the final lines with spacing.
        if i == 0:
            line1 += l1
            line2 += l2
            line3 += l3
            if show_answers:
                line4 += l4
        else:
            #add four spaces as a sperator between problems.
            line1 += '    ' + l1
            line2 += '    ' + l2
            line3 += '    ' + l3
            if show_answers:
                line4 += '    ' + l4
        
    # --- Final Assembly ---

    if show_answers:
        arranged_problems = line1 + '\n' + line2 + '\n' + line3 + '\n' + line4 
    else:
        arranged_problems = line1 + '\n' + line2 + '\n' + line3
    
    return arranged_problems
